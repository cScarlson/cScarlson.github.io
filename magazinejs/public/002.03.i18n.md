
Internationalization
================================================================

## Myth
_You need an i18n/l10n library in order to internationalize a site or app_.

## Reason
A fully scalable Translator service can be built in 25-50 LOC.

## Conventions
Assuming a couple of minor conventions, Internationalization/Localization becomes very simple. Let's assume we have `locale.{lang}.json` files that look about like this...

### `locale.en.json`
```json
{
    "PURCHASE:COMPLETE:MESSAGE": "Thank you for purchasing your new ${vehicle} ${locale['TAXON:DAY:TODAY']}!",
    "TAXON:VEHICLE:AUTO": "car",
    "TAXON:VEHICLE:MOTO": "motorcycle",
    "TAXON:DAY:TODAY": "today",
    "USER:SETTINGS:PRIVACY:BUTTON:SAVE": "Save",
    "USER:SETTINGS:PRIVACY:BUTTON:CANCEL": "Cancel",
    "USER:SETTINGS:PRIVACY:TAC": "${ltts.en}"
}
```

#### Conventions to Note
- **All** locale files use _fully-qualified_ translation keys -- there is no nesting as this radically complicates our translation system. Preferably, this follows a reasonable _Domain-Driven Design_.
- Advanced transltion values can use any convention of interpolation syntax.
- Interpolation tokens can use "_Taxa_", to automatically interpolate translation keys from the same locale file.
- Taxon translation keys should always start with the root qualifier of "`TAXON`".
- Translators can target specific keys or run the same operation over every process.

## `@core/utilities/l10n/translator.ts`
```typescript
const { LANG } = env;
const { location, navigator } = window;
const { searchParams } = new URL(location);
const { language: l, lang = LANG || searchParams.get('lang') || l } = navigator;
const LANGUAGE_EN_US = 'en-US';
const LANGUAGE_ES_MX = 'es-MX';
const DIALECT_EN = 'en';
const DIALECT_ES = 'es';
const DEFAULT_DIALECTS = {
    [LANGUAGE_EN_US]: DIALECT_EN,
    [LANGUAGE_ES_MX]: DIALECT_ES,
};

class Translator {
    language = LANGUAGE_EN_US;
    dialects = DEFAULT_DIALECTS;
    locales = { };
    translators = new Set();
    dialect = DEFAULT_DIALECTS[this.language];
    locale = this.locales[this.dialect];
    
    constructor(options = {}) {
        const { dialects, locales, translators } = { ...this, ...options };
        
        this.dialects = dialects;
        this.locales = locales;
        this.translators = translators;
        this.change(lang);
        
        return this;
    }
    
    change = (language = LANGUAGE_EN_US) => {
        const { dialects, locales } = this;
        const { [language]: dialect } = { ...DEFAULT_DIALECTS, ...dialects };
        const { [dialect]: locale } = locales;
        
        this.language = language;
        this.dialect = dialect;
        this.locale = locale;
        
        return dialect;
    };
    
    translate = (key, scope = {}) => {
        const { locale, translators, dialect } = this;
        const { [key]: value } = locale;
        return [ ...translators ].reduce( (value, fn) => fn.call({ dialect, key, locale, scope }, value), value );
    };
    
}

export { Translator };
```

## Usage: `@app/utilities/l10n/translator.ts`
```typescript
import { Translator as Core } from '@core/utilities/l10n';
export const { execute, change } = new (class Translator extends Core {})();
```

## Usage: `@app/children/**/*.component.ts`
```typescript
import { execute, change } from '@app/utilities/l10n';

export class UIElement extends HTMLElement {
    
    handleEvent(e: MouseEvent) {
        change('es');
        this.shadowRoot.innerHTML = this.render();
    }
    
    connectedCallback() {
        this.shadowRoot.addEventListener('click', this, true);
    }
    
    render() {
        const TITLE = translate('DOMAIN:SUBDOMAIN:TRANSLATION:KEY');
        return `<h1>${TITLE}</h1>`;
    }
};
```

## Usage: `@core/core/utilities/l10n/translators.ts`
```typescript
import { utilities } from '@core/utilities';

export const translators = new Set()
    .add(translateInterpolation)
    ;
export {
    translateInterpolation,
};

function translateInterpolation(value: string): string {
    const { key, locale, scope } = this;
    const context = Object.assign({}, scope, { locale });
    return utilities.interpolate(value)(context);
}
```

## Usage: `@app/core/utilities/l10n/translators.ts`
```typescript
import { marked } from 'marked';
import { utilities } from '@core/utilities';
import { translators as core } from '@core/utilities/l10n';
import { default as ltt_en } from '@app/children/**/TAC.en.md' with { type: 'text' };
import { default as ltt_es } from '@app/children/**/TAC.es.md' with { type: 'text' };

const translateMarkdown = {
    dialect: '',
    cache: '',
    
    call({ dialect, key, locale, scope }, value: string): string {
        if (key !== 'USER:SETTINGS:PRIVACY:TAC') return value;
        if (dialect === this.dialect) return this.cache;  // return cached result if dialect is the same as last run
        const context = Object.assign({}, scope, { locale });
        const ltts = { en: ltt_en, es: ltt_es };
        const text = utilities.interpolate(value)(ltts);  // interpolate the proper markdown content into translation value
        const interpolated = utilities.interpolate(text)(context);  // interpolate context variables into markdown
        
        this.dialect = dialect;
        this.cache = interpolated;
        
        return marked.parse(interpolated);  // parse as HTML
    }
    
};

export const translators = new Set([ ...core ])
    .add(translateMarkdown)
    ;
export {
    translateMarkdown,
};
```

## Synopsis
In virtually 40 LOC, we created a fully-scalable l10n system that performs the super simple mapping, employs a fallback system from injected environment variables (on `env`) to a `?lang` query parameter (which could also use `location.pathname.split('/').pop()`) to `navigator.language` to a `constructor` parameter, provides generic core _variable interpolation hooks_ including the use of **"TAXON" interpolation**, leverages specific translators that target specific translation keys for special operation only the app should know about such as which keys demand _Markdown Translations_, AKA LTTs (Large Text Translations). The `translators` system uses a specific flavor of the CoR pattern (_The Chain of Responsibility Pattern_) where, if you'd like to override a core translator, you can simply import it and call `translators.delete(name)` to remove it before adding your own. This does everything and more than any library out there that'll work until it doesn't. Others might allow nesting but that's simply a beartrap for the scalability of your code.
