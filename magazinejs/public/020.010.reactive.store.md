
Reactive Store
================

## Myth
_You need a library like Redux or otherwise to have a Reactive Store_.

## Reason
A Reactive Store can be accomplished in roughly 25 LOC using a `Subject` from _The Observer Pattern_ (see previous chapter).

### `@core/store/store.ts`
```typescript
import { type IObserver, Subject } from '@core/utilities/patterns/behavioral/observer';
import { type Action, Reducer } from './reducer';

export class Store<T = any> extends Subject {
    data = {};
    
    constructor(public data: T = {}, private reducer: Reducer) {
        super(data, 'data');
    }
    
    dispatch(action: Action) {
        const { reducer, data } = this;
        
        this.data = reducer.execute(data, action);
        if (this.data !== data) this.notify();
        
        return this;
    }
    
};
```

### `@core/store/reducer.ts`
```typescript
export type Action = { type: string, payload: any };
export class Reducer<T = any> {
    
    execute = (state: T, action: Action) => {
        const { type } = action;
        if (type in this) return this[type](state, action);
        return state;
    };
    
};
```

### Usage: `@app/core/store/store.ts`
```typescript
import { Store as Core } from '@core/store/store';
import { Reducer } from './reducer';
import { State } from './state';

const INITIAL_STATE = new State({ ... });
const reducer = new Reducer();

export const store = new (class Store extends Core {})(INITIAL_STATE, reducer);
```

### Usage: `@app/core/store/reducer.ts`
```typescript
import { type Action, Reducer as Core } from '@core/store/reducer';
import { State } from './state';
export class Reducer<T = State> extends Core<T> {
    
    execute = (state: T, action: Action) => {
        const latest = super.execute(state, action);
        const persisted = JSON.stringify(latest);
        
        localStorage.setItem('persisted', persisted);  // uses subclass as middleware
        return latest;
    };
    
};
```

### Usage: `@app/app.ts`
```typescript
import { type State, store } from '@app/store';

store.attach({
    call(state: State) { ... }
});
```

## Synopsis
By creating a store in about 15 LOC that extends a `Subject`, and leveraging a Reducer that is about 5 LOC, we accomplished a Reactive State Store that is reusable across any domain. In fact, we even leverage the Reducer's `execute` method to override it for specific needs.
