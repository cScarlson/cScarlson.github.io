
Poor Man's Components [Framework]
================
This is a demonstration of a quick & easy way to create a Custom Components framework.

## Basic
Here it is in its most terse form...
```typescript
class App {
    static tagName = 'app';
    static content = `<div>app<child></child></div>`;
}

class Child {
    static tagName = 'child';
    static content = `<div>child</div>`;
}

const modules = [
    App,
    Child,
];
const instances = modules.map( bootstrap.bind(document) );

function bootstrap(Module) {
    const { tagName, content } = Module;
    const elements = this.getElementsByTagName(tagName);
    const instances = [ ...elements ].reduce(mount.bind(Module), []);
    
    return instances;
}
    
function mount(instances, element) {
    const { content } = this;
    const instance = new this(element);
    const collection = [ ...instances, instance ];
    
    element.innerHTML = content;
    collection.push( ...modules.map( bootstrap.bind(element) ) );
    
    return collection;
}
```

## As a Framework

##### `./src/core/core.ts` (design)
```typescript
class Core {
    static $components: Map<string, any> = new Map();
    private Component: any = Object;
    private tagName: string = '';
    public target: HTMLElement = document.createElement('div');  // use default div
    get dataset() { return this.target.dataset }
    get classList() { return this.target.classList }
    
    constructor(options: { target: HTMLElement }&any = {}) {
        var options = { ...this, ...options };
        const { Component, tagName, target } = options;
        
        this.Component = Component;
        this.tagName = tagName;
        this.target = target;
        
        return this;
    }
    
    static bootstrap(Component: any) {
        const { tagName } = Component;
        const $targets = document.getElementsByTagName(tagName);
        const targets = Array.prototype.slice.call($targets);
        const instances = this.mount(Component, ...targets);
        
        return this;
    }
    
    static mount(Component: any, target?: HTMLElement, ...more: HTMLElement[]) { // Tail-Call Optimized
        if (!target) return Core.$components.get(Component.tagName);  // early-return
        const { $components } = Core;
        const { tagName } = Component;
        const $ = new Core({ Component, tagName, target });
        const instance = new Component($);
        
        if ( !$components.has(tagName) ) $components.set(tagName, [ ]);
        $components.get(tagName).push(instance);
        
        if (more.length) return this.mount(Component, ...more);
        return $components.get(tagName);
    }
    
    on(type: string, handler: EventListenerOrEventListenerObject, capture: boolean = true) {
        const { target } = this;
        target.addEventListener(type, handler, capture);
        return this;
    }
    
    off(type: string, handler: EventListenerOrEventListenerObject, capture: boolean = true) {
        const { target } = this;
        target.removeEventListener(type, handler, capture);
        return this;
    }
    
    fire(type: string, data?: any) {  // dispatches on host component
        const { target } = this;
        const e = new CustomEvent(type, { detail: data });
        
        target.dispatchEvent(e);
        return this;
    }
    
    load(content: string) {  // loads innerHTML from string
        const { target } = this;
        target.innerHTML = content;
        return this;
    }
    
    style(styles: string) {  // injects styles in document head if not already existing
        if ( document.head.querySelector(`#${this.tagName}`) ) return this;
        const { tagName } = this;
        const { head } = document;
        const style = document.createElement('style');
        
        style.id = tagName;
        style.innerHTML = styles;
        head.appendChild(style);
        
        return this;
    }
    
}

export { Core };
```

##### `./index.ts` (registration)
```typescript
import { Core } from './src/app/core';
import { App } from './src/app/app.component';
import { Child1 } from './src/app/subsystem/child1';
import { Child2 } from './src/app/subsystem/child2';

function bootstrap() {
    Core.bootstrap(App)
        .bootstrap(Child1)
        .bootstrap(Child2)
        ;
}

window.addEventListener('load', bootstrap, false);
if (document.readyState === 'complete') bootstrap();

```


##### `./src/app.component.ts`
```typescript
import { Core } from './core/Core';
import styles from './app.component.css';
import content from './app.component.html';

class App {
    static tagName: string = 'app';
    
    constructor(private $: Core) {
        $.style(styles).load(content);
    }
    
}

export { App };
```

## Synopsis
We basically just define a static member, "`tagName`", on our component definitions so that the Core can look them up in the DOM using `document.getElementsByTagName`. We add some convenience methods onto the core, although, we could have the engine take care of initial content loading automatically by defining a static "`content`" property on our component definitions (like `tagName`). As you can see, this [very] small system already grants us a lot of leverage. However, imagine going a step further to have `Core` inherit from, say, an "`EventBus`" module to provide PubSub communication across our components. Imagine wiring this up with a `MutationObserver`. You could even alter the way components are defined and|or registered/bootstrapped so that we can leverage a `Proxy` and use get/set modifications to serve components with objects based on a data-binding strategy. There's a lot we can do from here to flesh out a rather powerful framework. That said, this small system may be all you need for basic prototyping of UI's.
