
The Arbiter Pattern
================
The Arbiter Pattern is a JavaScript pattern invented by yours truly (me, Cody Carlson). It is a nifty little pattern to slicken your syntax. Say you have a library that you'd like to publish. Say this library ideally needs some syntactical sugar to sweeten up its usage. Here you go...


```javascript
class Core {
    
    method1() {}
    method2() {}  // private
    
}

var Facade = function Facade(core) {
    
    function method1(...args) {
        core.method1(...args);
        return this;
    }
    
    // export precepts
    this.method1 = method1;
    
    return this;
}

var V = new (function Vertex(Core, Facade) {
    var v = Facade.call(function v(param, ...args) {
        if (this instanceof v) return new Vertex(Core, Facade);  // creates new Core instance / API partitian.
        if (param instanceof String) return v.method1(param, ...args);  // decides what to do based upon the input signature
        // ... more conditions for signature
        
        return v;
    }, new Core());
    
    return v;
})(Core, Facade);

```

## Usage
```javascript
V('...');
V(1)(2)(3)('n');
V.method1('explicit');
V.method1('explicit')('implicit');

const app2 = new V('new instance of API');
app2('...');
app2(1)(2)(3)('n');
app2.method1('explicit');
app2.method1('explicit')('implicit');
```
